from __future__ import annotations

import logging
import os
import time

from typing import Any, AsyncGenerator

import anthropic
import httpx
from fastapi import FastAPI, HTTPException
from fastapi.middleware.cors import CORSMiddleware

from mcp.client.sse import sse_client

# ── Configuration ─────────────────────────────────────────────────────────────
ANTHROPIC_API_KEY = os.environ["ANTHROPIC_API_KEY"]
KALI_SERVER_URL   = os.getenv("KALI_SERVER_URL", "http://localhost:9090")
MCP_SERVER_URL    = os.getenv("MCP_SERVER_URL",  "http://localhost:5001/sse")
HITL_SECRET       = os.environ["HITL_SECRET"]
CLAUDE_MODEL      = "claude-sonnet-4-20250514"
DESTRUCTIVE_TOOLS = {"sqlmap_scan", "metasploit_run"}

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s [%(levelname)s] %(name)s — %(message)s",
)
log = logging.getLogger("kali-gateway")


# ── App state ─────────────────────────────────────────────────────────────────
class AppState:
    mcp_tools: list[dict[str, Any]] = []
    anthropic_client: anthropic.AsyncAnthropic | None = None
    http_client: httpx.AsyncClient | None = None

state = AppState()


# ── Lifespan ──────────────────────────────────────────────────────────────────
@asynccontextmanager
async def lifespan(app: FastAPI):
    log.info("Gateway starting up …")
    state.anthropic_client = anthropic.AsyncAnthropic(api_key=ANTHROPIC_API_KEY)
    state.http_client = httpx.AsyncClient(base_url=KALI_SERVER_URL, timeout=600)

    try:
        state.mcp_tools = await fetch_mcp_tools()
        log.info("Loaded %d MCP tools: %s",
                 len(state.mcp_tools), [t["name"] for t in state.mcp_tools])
    except Exception as exc:
        log.warning("Could not reach MCP server at startup: %s", exc)

    yield
    await state.http_client.aclose()
    log.info("Gateway shut down.")


# ── FastAPI app ───────────────────────────────────────────────────────────────
app = FastAPI(
    title="Kali AI Agent Gateway",
    version="1.0.0",
    description="AI-Powered Pentesting Agent — Claude + Kali Linux tools. Test via /docs",
    lifespan=lifespan,
    docs_url="/docs",
    redoc_url="/redoc",
)

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)


# ── Models ────────────────────────────────────────────────────────────────────
class ChatMessage(BaseModel):
    role: str
    content: str

class ChatRequest(BaseModel):
    messages: list[ChatMessage]
    hitl_approved: dict[str, str] = {}

    @field_validator("messages")
    @classmethod
    def must_have_messages(cls, v):
        if not v:
            raise ValueError("messages must not be empty")
        return v


# ── MCP tool discovery ────────────────────────────────────────────────────────
async def fetch_mcp_tools() -> list[dict[str, Any]]:
    async with sse_client(MCP_SERVER_URL) as (read, write):
        async with ClientSession(read, write) as session:
            await session.initialize()
            tools_result = await session.list_tools()

    return [
        {
            "name": tool.name,
            "description": tool.description or f"Run {tool.name}",
            "input_schema": tool.inputSchema or {"type": "object", "properties": {}},
        }
        for tool in tools_result.tools
    ]


# ── HITL token ────────────────────────────────────────────────────────────────
def generate_hitl_token(job_id: str) -> tuple[str, str]:
    ts = int(time.time())
    message = f"hitl:{job_id}:{ts}"
    token = hmac.new(
        HITL_SECRET.encode(), message.encode(), hashlib.sha256
    ).hexdigest()
    return token, f"{job_id}:{ts}"


# ── Go server caller — waits for full scan output ─────────────────────────────
async def call_go_server(
    tool_name: str,
    tool_input: dict[str, Any],
    hitl_approved: dict[str, str],
) -> dict[str, Any]:

    endpoint_map = {
        "nmap_scan":    "/tool/nmap",
        "nikto_scan":   "/tool/nikto",
        "ffuf_fuzz":    "/tool/ffuf",
        "sqlmap_scan":  "/tool/sqlmap",
        "dns_lookup":   "/tool/dns",
    }
    endpoint = endpoint_map.get(tool_name)
    if not endpoint:
        raise HTTPException(400, f"Unknown tool: {tool_name}")

    headers: dict[str, str] = {"Content-Type": "application/json"}

    if tool_name in DESTRUCTIVE_TOOLS:
        approved = hitl_approved.get(tool_name)
        if not approved:
            raise HTTPException(
                403,
                f"Tool '{tool_name}' requires Human-in-the-Loop approval.",
            )
        token, payload = approved.split("|", 1)
        headers["X-HITL-Token"]   = token
        headers["X-HITL-Payload"] = payload

    # Step 1 — start the job, get job_id back immediately
    response = await state.http_client.post(
        endpoint, json=tool_input, headers=headers
    )
    response.raise_for_status()
    job = response.json()
    job_id = job.get("job_id")

    if not job_id:
        return job

    log.info("Job %s started for tool %s — collecting output stream", job_id, tool_name)

    # Step 2 — connect to SSE stream and collect ALL output lines
    output_lines: list[str] = []
    stream_url = f"{KALI_SERVER_URL}/stream/{job_id}"

    try:
        async with httpx.AsyncClient(timeout=300) as stream_client:
            async with stream_client.stream("GET", stream_url) as stream:
                async for line in stream.aiter_lines():
                    if not line.startswith("data:"):
                        continue
                    raw = line[5:].strip()
                    if not raw:
                        continue
                    try:
                        event = json.loads(raw)
                        etype = event.get("type", "")
                        data  = event.get("data", "")

                        if etype == "stdout":
                            output_lines.append(data)
                        elif etype == "stderr":
                            output_lines.append(f"[stderr] {data}")
                        elif etype in ("exit", "error", "close"):
                            break
                    except Exception:
                        continue
    except Exception as exc:
        log.error("Stream collection failed for job %s: %s", job_id, exc)

    full_output = "\n".join(output_lines) if output_lines else "No output received"
    log.info("Job %s complete — %d lines collected", job_id, len(output_lines))

    # Step 3 — return full scan output so Claude can read and explain it
    return {
        "job_id":  job_id,
        "tool":    tool_name,
        "target":  tool_input.get("target", ""),
        "output":  full_output,
    }


# ── System prompt ─────────────────────────────────────────────────────────────
SYSTEM_PROMPT = """You are KaliAgent, an expert penetration tester and security analyst.
You have access to Kali Linux tools. Your workflow:

1. Analyse the user request and pick the right tool.
2. Call the tool with precise arguments.
3. When you receive the scan output, provide a structured analysis:
   - Executive summary
   - Open ports / findings
   - Risk level (Critical / High / Medium / Low / Info)
   - Recommended next steps

RULES:
- Only scan targets you have explicit permission to test.
- Flag destructive tools (sqlmap) before calling — the system will pause for approval.
- Prefer recon tools (nmap, nikto) before exploitation tools.
"""


# ── Agent loop ────────────────────────────────────────────────────────────────
async def agent_stream(
    messages: list[dict],
    hitl_approved: dict[str, str],
) -> AsyncGenerator[str, None]:

    if not state.mcp_tools:
        try:
            state.mcp_tools = await fetch_mcp_tools()
        except Exception as exc:
            yield f"data: {json.dumps({'type':'error','content':str(exc)})}\n\n"
            return

    conversation = list(messages)
    max_rounds = 6

    for round_num in range(max_rounds):
        log.info("Agent round %d — %d messages in context", round_num + 1, len(conversation))

        response = await state.anthropic_client.messages.create(
            model=CLAUDE_MODEL,
            max_tokens=4096,
            system=SYSTEM_PROMPT,
            tools=state.mcp_tools,
            messages=conversation,
        )

        # Stream text to client
        for block in response.content:
            if block.type == "text":
                yield f"data: {json.dumps({'type':'text','content':block.text})}\n\n"

        if response.stop_reason == "end_turn":
            yield f"data: {json.dumps({'type':'done'})}\n\n"
            return

        if response.stop_reason != "tool_use":
            yield f"data: {json.dumps({'type':'done','reason':response.stop_reason})}\n\n"
            return

        # Handle tool calls
        tool_results = []
        for block in response.content:
            if block.type != "tool_use":
                continue

            tool_name  = block.name
            tool_input = block.input
            tool_id    = block.id

            log.info("Claude requested tool: %s  input: %s", tool_name, tool_input)
            yield f"data: {json.dumps({'type':'tool_call','tool':tool_name,'input':tool_input})}\n\n"

            # HITL gate for destructive tools
            if tool_name in DESTRUCTIVE_TOOLS and tool_name not in hitl_approved:
                token, payload = generate_hitl_token(tool_id)
                yield f"data: {json.dumps({'type':'hitl_required','tool':tool_name,'token':token,'payload':payload})}\n\n"
                yield f"data: {json.dumps({'type':'done','reason':'awaiting_hitl'})}\n\n"
                return

            # Call Go server and collect full output
            try:
                result = await call_go_server(tool_name, tool_input, hitl_approved)
                tool_results.append({
                    "type":        "tool_result",
                    "tool_use_id": tool_id,
                    "content":     json.dumps(result),
                })
                yield f"data: {json.dumps({'type':'tool_result','tool':tool_name,'result':result})}\n\n"
            except Exception as exc:
                log.error("Tool %s failed: %s", tool_name, exc)
                tool_results.append({
                    "type":        "tool_result",
                    "tool_use_id": tool_id,
                    "content":     f"Error: {exc}",
                    "is_error":    True,
                })

        conversation.append({"role": "assistant", "content": response.content})
        conversation.append({"role": "user",      "content": tool_results})

    yield f"data: {json.dumps({'type':'error','content':'Max agent rounds reached'})}\n\n"


# ── Endpoints ─────────────────────────────────────────────────────────────────
@app.get("/health")
async def health():
    return {"status": "ok", "service": "kali-gateway", "model": CLAUDE_MODEL}


@app.get("/tools")
async def list_tools():
    if not state.mcp_tools:
        state.mcp_tools = await fetch_mcp_tools()
    return {"tools": state.mcp_tools}


@app.post("/chat")
async def chat(req: ChatRequest):
    """
    Main streaming endpoint. Returns Server-Sent Events.
    Event types: text | tool_call | tool_result | hitl_required | done | error
    """
    messages = [m.model_dump() for m in req.messages]

    async def event_generator():
        try:
            async for chunk in agent_stream(messages, req.hitl_approved):
                yield chunk
        except Exception as exc:
            log.exception("Unhandled error in agent_stream")
            yield f"data: {json.dumps({'type':'error','content':str(exc)})}\n\n"

    return StreamingResponse(
        event_generator(),
        media_type="text/event-stream",
        headers={
            "Cache-Control":    "no-cache",
            "X-Accel-Buffering":"no",
        },
    )
